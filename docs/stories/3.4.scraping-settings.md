## Story
**As a** usuário,
**I want** acionar novas buscas e configurar parâmetros de scraping diretamente pelo dashboard,
**so that** eu possa operar tudo sem usar terminal e gerenciar minhas credenciais de API.

## Acceptance Criteria
1. Modal "Nova Busca" com campos: plataforma (dropdown), keyword/nicho (input), quantidade (slider 10-100)
2. Botão "Buscar Prospects" chama `POST /api/scrape` e exibe loader durante execução
3. Ao finalizar, tabela de leads atualiza automaticamente com novos leads
4. Página `/settings` exibe campos para inserir/atualizar `APIFY_TOKEN` e `GEMINI_API_KEY` (valores mascarados)
5. Configurações salvas em `.env.local` via endpoint `POST /api/settings`

### Componente ScrapeModal
[Source: docs/architecture.md#frontend-architecture]
```typescript
// src/components/leads/ScrapeModal.tsx
"use client";

import { useState } from "react";
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import type { Platform } from "@/types/lead";

interface ScrapeModalProps {
  open: boolean;
  onClose: () => void;
  onScrapeComplete: () => void;
}

export function ScrapeModal({ open, onClose, onScrapeComplete }: ScrapeModalProps) {
  const [platform, setPlatform] = useState<Platform>("instagram");
  const [keyword, setKeyword] = useState("");
  const [limit, setLimit] = useState(10);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleScrape = async () => {
    if (!keyword.trim()) {
      setError("Digite uma palavra-chave");
      return;
    }
    if (limit < 10 || limit > 100) {
      setError("Limite entre 10 e 100");
      return;
    }

    setLoading(true);
    setError(null);
    try {
      const res = await fetch("/api/scrape", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ platform, keyword, limit }),
      });
      if (!res.ok) {
        const { error } = await res.json();
        setError(error ?? "Falha no scraping");
        return;
      }
      const result = await res.json();
      // Success feedback
      alert(`Importados: ${result.imported}, Ignorados: ${result.skipped}`);
      onScrapeComplete();
      onClose();
    } catch {
      setError("Falha de conexão");
    } finally {
      setLoading(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Buscar Novos Prospects</DialogTitle>
        </DialogHeader>
        <div className="space-y-4">
          <div>
            <label className="text-sm font-semibold">Plataforma</label>
            <Select value={platform} onValueChange={setPlatform as any}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="instagram">Instagram</SelectItem>
                <SelectItem value="linkedin">LinkedIn</SelectItem>
              </SelectContent>
            </Select>
          </div>

          <div>
            <label className="text-sm font-semibold">Palavra-chave</label>
            <Input
              value={keyword}
              onChange={e => setKeyword(e.target.value)}
              placeholder="Ex: clínica odontológica"
            />
          </div>

          <div>
            <label className="text-sm font-semibold">Quantidade ({limit})</label>
            <input
              type="range"
              min={10}
              max={100}
              value={limit}
              onChange={e => setLimit(Number(e.target.value))}
              className="w-full"
            />
          </div>

          {error && <p className="text-red-600 text-sm">{error}</p>}

          <Button onClick={handleScrape} disabled={loading} className="w-full">
            {loading ? "Buscando..." : "Buscar Prospects"}
          </Button>
        </div>
      </DialogContent>
    </Dialog>
  );
}
```

### POST /api/settings
[Source: docs/architecture.md#backend-architecture]
```typescript
// src/app/api/settings/route.ts
import { NextRequest, NextResponse } from "next/server";
import { writeFile, readFile } from "fs/promises";
import { existsSync } from "fs";

const ENV_LOCAL = ".env.local";

export async function POST(request: NextRequest) {
  try {
    const { APIFY_TOKEN, GEMINI_API_KEY } = await request.json();

    if (!APIFY_TOKEN?.trim() || !GEMINI_API_KEY?.trim()) {
      return NextResponse.json({ error: "Chaves não podem ser vazias" }, { status: 400 });
    }

    // Ler .env.local existente
    let envContent = "";
    if (existsSync(ENV_LOCAL)) {
      envContent = await readFile(ENV_LOCAL, "utf-8");
    }

    // Atualizar ou adicionar chaves
    const lines = envContent.split("\n");
    const updated = new Set(["APIFY_TOKEN", "GEMINI_API_KEY"]);
    let newContent = lines
      .filter(line => {
        const key = line.split("=")[0];
        if (updated.has(key)) {
          updated.delete(key);
          return false;
        }
        return true;
      })
      .join("\n");

    newContent += `\nAPIFY_TOKEN=${APIFY_TOKEN}\nGEMINI_API_KEY=${GEMINI_API_KEY}`;
    await writeFile(ENV_LOCAL, newContent, "utf-8");

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error("[POST /api/settings]", error);
    return NextResponse.json({ error: "Falha ao salvar" }, { status: 500 });
  }
}
```
